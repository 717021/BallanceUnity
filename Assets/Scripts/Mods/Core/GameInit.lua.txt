GameBulider = Ballance2.GameBulider
GamePathManager = Ballance2.Utils.GamePathManager

GameModLoader = GameBulider.GetGameService(typeof(Ballance2.GameModLoader), GameServices.GameModLoader)

loader_err = ""
loader_succeed = false

function GameInit_GetLoaderSucceed() return loader_succeed end
function GameInit_GetLoaderError() return loader_err end
function GameInit_Entry()
  return StartCoroutine(GameInit_Coroutine)
end
function GameInit_Coroutine()
  GameMgr.Log("Gameinit script start")
  GameMgr.Log("Game version : {0} Bulid version : {1}", GameConst.GameVersion, GameConst.GameBulidVersion)
  GameMgr.Log("Gameinit finished")
end
